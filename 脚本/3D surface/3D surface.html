x <- seq(-4, 4, 0.1)
y <- seq(-4, 4, 0.1)
sc <- function(x, y) {
    sin(x) * cos(y)
}
z <- outer(x, y, sc)

color <- colorRampPalette(c("blue", "green"))(100)
zfacet <- z[-4, -4] + z[-4, -ncol(z)] + z[-nrow(z), -4] + z[-nrow(z), -ncol(z)]
facetcol <- cut(zfacet, 100)

persp(x, y, z, phi = 45, theta = -30, col = color[facetcol])
